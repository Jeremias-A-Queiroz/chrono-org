(defun jm/org-log-manual-clock (start end)
  "Transcribes an external time record to LOGBOOK natively and safely."
  (interactive "sStart (YYYY-MM-DD HH:MM): \nsEnd (YYYY-MM-DD HH:MM): ")
  (let ((start-ts (org-read-date nil t start))
        (end-ts (org-read-date nil t end)))
    (save-excursion
      (org-back-to-heading t)
      (org-clock-in nil start-ts)
      (org-clock-out nil nil end-ts)
      (org-clock-update-time-maybe)
      (message "Time record inserted in %s." (buffer-name)))))

(local-set-key (kbd "C-c L") #'jm/org-log-manual-clock)
#+end_src
